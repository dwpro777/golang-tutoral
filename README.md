<h1>Getting up and running: </h1> 
export GOPATH=$HOME/.go  
sudo dnf install golang  
sudo dnf install *golang*sqlite*  
go get github.com/mattn/go-sqlite3  

<h1>Optional:  </h1> 
sudo dnf install gitk  
sudo dnf install sqliteman
